### Opción 2: Queries Útiles

```sql
-- Comparar afinidades L vs F para ligandos comunes
SELECT 
    L.ligand_name,
    L.affinity_kcal_mol AS L_affinity,
    F.affinity_kcal_mol AS F_affinity,
    (L.affinity_kcal_mol - F.affinity_kcal_mol) AS selectivity
FROM docking_results_L L
JOIN docking_results_F_1144 F ON L.ligand_name = F.ligand_name
WHERE L.affinity_kcal_mol < -8.0  -- Excelentes en L
ORDER BY selectivity DESC;

-- Top 50 ligandos de L
SELECT ligand_name, affinity_kcal_mol 
FROM docking_results_L 
WHERE affinity_kcal_mol < -8.0
ORDER BY affinity_kcal_mol 
LIMIT 50;

-- Distribución de afinidades
SELECT 
    CASE 
        WHEN affinity_kcal_mol < -8.0 THEN 'Excelente'
        WHEN affinity_kcal_mol < -7.0 THEN 'Buena'
        WHEN affinity_kcal_mol < -6.0 THEN 'Moderada'
        WHEN affinity_kcal_mol < -5.0 THEN 'Débil'
        ELSE 'Muy débil'
    END AS categoria,
    COUNT(*) as cantidad
FROM docking_results_L
GROUP BY categoria;
```
